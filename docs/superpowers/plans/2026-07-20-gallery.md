# Gallery Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a standalone `gallery.html` page that lists every diagram from the deep-dive companions, shows curated context and a "why" summary, and links back to each deep dive; enforce maintenance via `gallery-registry.json` and `scripts/validate.py`.

**Architecture:** A Python generator script parses all `deep-dives/*.html` files, extracts `<figure>` blocks, matches them to a human-curated JSON registry by `{filename_stem, figure_index}`, and emits `gallery.html`. The existing validator is extended to fail if any figure lacks a registry entry or any registry entry is orphaned.

**Tech Stack:** Python 3 (stdlib only: `html.parser`, `json`, `re`, `os`), hand-authored HTML/CSS matching the existing deep-dive visual style.

## Global Constraints

- Zero runtime dependencies and zero build pipeline for the front-end artifact.
- `gallery.html` is generated and committed; `scripts/generate-gallery.py` is the source of truth for its structure.
- Every `<figure>` in `deep-dives/*.html` must have a matching entry in `gallery-registry.json`.
- All SVG ids inlined into `gallery.html` must remain globally unique.
- Follow existing file naming, CSS token names, and the overlay/deep-dive visual vocabulary.
- Run `python3 scripts/validate.py` after every change; it must stay green.
- Working branch: `feature/gallery`.

## File Structure

| File | Responsibility |

```python
#!/usr/bin/env python3
"""Generate gallery.html from deep-dive figures and a curated registry."""
import json
import os
import re
from html.parser import HTMLParser

ROOT = os.path.dirname(os.path.dirname(os.path.abspath(__file__)))
DEEPDIR = os.path.join(ROOT, "deep-dives")
REGISTRY = os.path.join(ROOT, "gallery-registry.json")
OUT = os.path.join(ROOT, "gallery.html")


class FigureParser(HTMLParser):
    def __init__(self):
        super().__init__()
        self.figures = []
        self._in_figure = False
        self._in_figcaption = False
        self._in_h2 = False
        self._current_h2_text = ""
        self._current_h2_anchor = None
        self._current_caption = ""
        self._current_section = ""
        self._current_section_anchor = None

    def handle_starttag(self, tag, attrs):
        attrs = dict(attrs)
        if tag == "figure":
            self._in_figure = True
            self._current_caption = ""
            self._current_section = self._current_h2_text
            self._current_section_anchor = self._current_h2_anchor
        elif tag == "figcaption" and self._in_figure:
            self._in_figcaption = True
        elif tag == "h2":
            self._in_h2 = True
            self._current_h2_text = ""
            self._current_h2_anchor = attrs.get("id")
        # Ignore any nested tags inside figcaption; we only collect text.

    def handle_endtag(self, tag):
        if tag == "figure" and self._in_figure:
            self.figures.append({
                "caption": self._current_caption.strip(),
                "section": self._current_section.strip(),
                "section_anchor": self._current_section_anchor,
            })
            self._in_figure = False
            self._in_figcaption = False
        elif tag == "figcaption":
            self._in_figcaption = False
        elif tag == "h2":
            self._in_h2 = False

    def handle_data(self, data):
        if self._in_h2:
            self._current_h2_text += data
        elif self._in_figcaption:
            self._current_caption += data


def extract_figures(html_path):
    parser = FigureParser()
    with open(html_path, encoding="utf-8") as f:
        parser.feed(f.read())
    return parser.figures


def deep_dive_stem(filename):
    """Return filename without .html extension."""
    return filename[:-5] if filename.endswith(".html") else filename


def draft_registry(deep_dir, out_path):
    entries = []
    for fn in sorted(os.listdir(deep_dir)):
        if not fn.endswith(".html"):
            continue
        stem = deep_dive_stem(fn)
        figures = extract_figures(os.path.join(deep_dir, fn))
        for idx, fig in enumerate(figures):
            entries.append({
                "deepDive": stem,
                "figureIndex": idx,
                "title": fig["caption"] or f"Diagram in {stem}",
                "section": fig["section"] or "",
                "sectionAnchor": fig["section_anchor"] or "",
                "why": "",
            })
    with open(out_path, "w", encoding="utf-8") as f:
        json.dump({"entries": entries}, f, indent=2, ensure_ascii=False)
        f.write("\n")
    print(f"Wrote {len(entries)} draft entries to {out_path}")


if __name__ == "__main__":
    import sys
    if len(sys.argv) > 1 and sys.argv[1] == "--draft-registry":
        draft_registry(DEEPDIR, os.path.join(ROOT, "gallery-registry-draft.json"))
    else:
        print("Usage: python3 scripts/generate-gallery.py --draft-registry")
        print("       python3 scripts/generate-gallery.py")
```

```python
SVG_RE = re.compile(r"<svg\b.*?</svg>", re.S)

def extract_svgs(html_path):
    with open(html_path, encoding="utf-8") as f:
        s = f.read()
    return SVG_RE.findall(s)


def collect_svg_styles(deep_dir):
    """Gather all standalone deep-dive SVG edge/marker CSS rules.

    Deep-dive files add per-file rules like:
      svg .edE1{stroke:#57656F;...;marker-end:url(#ahE1a)}
    The gallery inlines many SVGs, so it needs the union of these rules.
    """
    rules = []
    for fn in sorted(os.listdir(deep_dir)):
        if not fn.endswith(".html"):
            continue
        with open(os.path.join(deep_dir, fn), encoding="utf-8") as f:
            text = f.read()
        style = re.search(r"<style[^>]*>(.*?)</style>", text, re.S)
        if not style:
            continue
        css = style.group(1)
        # Capture any svg .ed* or svg marker-related rule.
        # Single-class rules: svg .ed { ... } and svg .edE1 { ... }
        for rule in re.findall(r"svg\s+\.ed(?:[\w-]+)?\s*\{[^}]+\}", css):
            if rule not in rules:
                rules.append(rule)
        # Grouped selectors, e.g. svg .ed, svg .edD { ... }
        for rule in re.findall(r"svg\s+\.ed(?:[\w-]+)?(?:\s*,\s*svg\s+\.ed[\w-]+)*\s*\{[^}]+\}", css):
            if rule not in rules:
                rules.append(rule)
        # Marker defs styling, e.g. #ahE1a polygon { ... } or marker polygon { ... }
        for rule in re.findall(r"(?:#ah[\w]+|marker)\s+[\w,\s]+\{[^}]+\}", css):
            if rule not in rules:
                rules.append(rule)
    return "\n".join(rules)
```

```python
def check_gallery_registry():
    print("gallery registry")
    registry = load_registry()
    if registry is None:
        return

    entries = registry.get("entries", [])
    if not isinstance(entries, list):
        note(False, "gallery-registry.json: 'entries' must be a list")
        return

    # Schema check
    keys = set()
    dup_keys = set()
    required = {"deepDive", "figureIndex", "why"}
    for i, e in enumerate(entries):
        missing = required - set(e.keys())
        if missing:
            note(False, f"gallery-registry.json entry {i} missing fields: {sorted(missing)}")
        if not isinstance(e.get("deepDive"), str):
            note(False, f"gallery-registry.json entry {i}: deepDive must be a string")
        if not isinstance(e.get("figureIndex"), int):
            note(False, f"gallery-registry.json entry {i}: figureIndex must be an int")
        if not isinstance(e.get("why"), str):
            note(False, f"gallery-registry.json entry {i}: why must be a string")
        if isinstance(e.get("why"), str) and not e.get("why").strip():
            note(False, f"gallery-registry.json entry {i}: why is empty")
        key = (e.get("deepDive"), e.get("figureIndex"))
        if key in keys:
            dup_keys.add(key)
        keys.add(key)
    if dup_keys:
        note(False, f"gallery-registry.json duplicate keys: {sorted(dup_keys)}")

    # Coverage check: every figure in deep-dives has a registry entry.
    deep_figures = {}
    for fn in sorted(os.listdir(DEEPDIR)):
        if not fn.endswith(".html"):
            continue
        stem = fn[:-5]
        s = open(os.path.join(DEEPDIR, fn), encoding="utf-8").read()
        count = len(re.findall(r"<figure\b", s))
        deep_figures[stem] = count

    registry_figures = {}
    for e in entries:
        registry_figures.setdefault(e["deepDive"], set()).add(e["figureIndex"])

    for stem, count in sorted(deep_figures.items()):
        expected = set(range(count))
        actual = registry_figures.get(stem, set())
        missing = expected - actual
        if missing:
            note(False, f"{stem}.html missing gallery-registry entries for figureIndex: {sorted(missing)}")
        extra = actual - expected
        if extra:
            note(False, f"gallery-registry.json has orphaned figureIndex for {stem}.html: {sorted(extra)}")

    # Orphaned deepDive values pointing to non-existent files.
    for stem in sorted(registry_figures.keys() - deep_figures.keys()):
        note(False, f"gallery-registry.json references missing deep-dive file: {stem}.html")

    note(True, f"gallery-registry.json covers all {sum(deep_figures.values())} figures")


def check_svg_id_uniqueness():
    print("svg id uniqueness")
    ids_by_file = {}
    dupes = {}
    for fn in sorted(os.listdir(DEEPDIR)):
        if not fn.endswith(".html"):
            continue
        s = open(os.path.join(DEEPDIR, fn), encoding="utf-8").read()
        ids = re.findall(r'<[^>]+\bid="([^"]+)"', s)
        for i in ids:
            if i in ids_by_file and ids_by_file[i] != fn:
                dupes.setdefault(i, set()).update([ids_by_file[i], fn])
            ids_by_file[i] = fn
    if dupes:
        note(False, f"SVG ids reused across deep dives (would collide in gallery.html): {sorted(dupes)}")
    else:
        note(True, "all SVG ids are unique across deep-dives")
```
